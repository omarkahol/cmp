84 // Fast bit-mixer to generate deterministic pseudo-randomness based on bit history
85 static inline uint64_t hash_prefix(uint64_t prefix, uint64_t seed) {
86 uint64_t v = prefix ^ seed;
87 v ^= v >> 30;
88 v *= 0xbf58476d1ce4e5b9ULL;
89 v ^= v >> 27;
90 v *= 0x94d049bb133111ebULL;
91 v ^= v >> 31;
92 return v;
93 }
94
95 // True stateless nested Owen Scrambling
96 static inline uint64_t owen_scramble(uint64_t val, uint64_t seed) {
97 uint64_t result = 0;
98 for(int i = 0; i < 64; ++i) {
99 int shift = 63 - i;
100
101 // The permutation of this bit depends ONLY on the preceding bits (the prefix)
102 uint64_t prefix = (i == 0) ? 0 : (val >> (shift + 1));
103
104 // Hash the prefix and seed to decide whether to flip the current bit
105 uint64_t flip = hash_prefix(prefix, seed) & 1;
106 uint64_t bit = ((val >> shift) & 1) ^ flip;
107
108 result = (result << 1) | bit;
109 }
110 return result;
111 }

113 public:
114 ScrambledSobolGrid(const Eigen::Ref<const Eigen::VectorXd> &lowerBound,

118 lowerBound_(lowerBound), upperBound_(upperBound) {

138 Eigen::VectorXd point(dimension_);
139 std::vector<boost::uint_least64_t> seq(dimension_);
140
141 gen_.generate(seq.begin(), seq.end());
142 double max_val = static_cast<double>(gen_.max());
143
144 for(size_t d = 0; d < dimension_; ++d) {
145 // 1. Normalize the boost integer to [0, 1)
146 double u = static_cast<double>(seq[d]) / max_val;
147
148 // 2. Map into a full 64-bit integer representing the fractional bits
149 // (Bit 63 is the 0.5 space cut, bit 62 is 0.25, etc.)
150 // Using 18446744073709551615.0 (which is 2^64 - 1)
151 uint64_t fractional_bits = static_cast<uint64_t>(u * 18446744073709551615.0);
152
153 // 3. Apply the true Owen scrambling algorithm
154 uint64_t scrambled_bits = owen_scramble(fractional_bits, dim_seeds_[d]);
155
156 // 4. Map back to [0, 1] float space
157 double scrambled_u = static_cast<double>(scrambled_bits) / 18446744073709551615.0;
158
159 // 5. Scale to final parameter bounds
160 point(d) = lowerBound_(d) + (upperBound_(d) - lowerBound_(d)) * scrambled_u;
161 }
162 return point;
