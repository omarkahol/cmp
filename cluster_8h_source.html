154 void updateAlpha() {
155 double K = static_cast<double>(clusters_.size());
156 double N = static_cast<double>(nPoints_);
157
158 // Extract prior parameters from your distribution object
159 double a = alphaPrior_.getAlpha();
160 double b = alphaPrior_.getBeta();
161
162 // 1. Instantiate the Beta distribution and sample eta
163 cmp::distribution::BetaDistribution beta_dist(alpha_ + 1.0, N);
164 double eta = beta_dist.sample(rng_);
165
166 // 2. Calculate the weights for the Gamma mixture
167 double weight1 = (a + K - 1.0) / (N * (b - std::log(eta)));
168 double pi_eta = weight1 / (weight1 + 1.0);
169
170 // 3. Sample the new alpha using your custom cmp::distribution classes
171 double u = distU_(rng_);
172 double updated_beta = b - std::log(eta); // The rate parameter
173
174 if(u < pi_eta) {
175 // Gamma(a + K, b - ln(eta))
176 cmp::distribution::GammaDistribution new_gamma(a + K, updated_beta);
177 alpha_ = new_gamma.sample(rng_);
178 } else {
179 // Gamma(a + K - 1, b - ln(eta))
180 cmp::distribution::GammaDistribution new_gamma(a + K - 1.0, updated_beta);
181 alpha_ = new_gamma.sample(rng_);
182 }
183
184 // Safety bounds just in case of severe numerical underflow
185 if(alpha_ < 1e-5) alpha_ = 1e-5;
186 }
