18 Eigen::VectorXd firstOrder; // First order indices

21 // Stores the interaction indices for second order if computed
22 // First element is interaction between 0 and 1, second between 0 and 2, ..., (d-1) and d, etc.

24
25 double getFirstOrder(size_t i) const {
26 if(i >= firstOrder.size()) throw std::out_of_range("Index out of range");
27 return firstOrder(i);
28 }
29 double getTotalOrder(size_t i) const {
30 if(i >= totalOrder.size()) throw std::out_of_range("Index out of range");
31 return totalOrder(i);
32 }
33 double getSecondOrder(size_t i, size_t j) const {
34 if(secondOrder.size() == 0) throw std::logic_error("Second order Sobol indices were not computed.");
35 if(i >= firstOrder.size() || j >= firstOrder.size()) throw std::out_of_range("Index out of range");
36 if(i == j) throw std::invalid_argument("Indices must be different for second order Sobol index.");
37 if(i > j) std::swap(i, j);
38 size_t index = i * (2 * firstOrder.size() - i - 1) / 2 + (j - i - 1);
39 return secondOrder(index);
40 }
41};
