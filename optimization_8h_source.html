38 // Call operator evaluated by NLoptCallback
39 double operator()(Eigen::Ref<const Eigen::VectorXd> x_opt, Eigen::Ref<Eigen::VectorXd> grad_opt) const {
40 // 1. Transform optimizer-space parameters to physical real-space parameters
41 Eigen::VectorXd x_real = mapToReal(x_opt);
42
43 double val;
44 if(use_gradient_ && grad_opt.size() > 0) {
45 // 2. Evaluate the user's function with real parameters
46 val = fval_grad_inplace_(x_real, grad_opt);
47 // 3. Chain Rule: d/dx_opt = d/dx_real * dx_real/dx_opt
48 mapGradientToOpt(x_real, grad_opt);
49 } else if(use_gradient_) {
50 Eigen::VectorXd dummy_grad(x_opt.size());

54 }

117 // Helper: Map optimizer space x (log) to real physical space x
118 Eigen::VectorXd mapToReal(Eigen::Ref<const Eigen::VectorXd> x_opt) const {
119 if(log_scale_.empty()) return x_opt;
120 Eigen::VectorXd x_real = x_opt;
121 for(int i = 0; i < x_real.size(); ++i) {
122 if(i < (int)log_scale_.size() && log_scale_[i]) {
123 x_real(i) = std::exp(x_opt(i));
124 }
125 }
126 return x_real;
127 }

129 // Helper: Map real gradient back to optimizer space gradient via Chain Rule
130 void mapGradientToOpt(Eigen::Ref<const Eigen::VectorXd> x_real, Eigen::Ref<Eigen::VectorXd> grad_real) const {
131 if(log_scale_.empty() || grad_real.size() == 0) return;
132 for(int i = 0; i < grad_real.size(); ++i) {
133 if(i < (int)log_scale_.size() && log_scale_[i]) {
134 // If x_real = exp(x_opt), then d(x_real)/d(x_opt) = exp(x_opt) = x_real
135 grad_real(i) = grad_real(i) * x_real(i);
136 }
137 }

140 // Helper to evaluate constraints dynamically while respecting log scaling
141 double evaluateConstraint(size_t index, bool is_ineq, Eigen::Ref<const Eigen::VectorXd> x_opt, Eigen::Ref<Eigen::VectorXd> grad_opt) const {
142 Eigen::VectorXd x_real = mapToReal(x_opt);
143 double val = 0.0;
144
145 const auto& constraint_func = is_ineq ? ineq_constraints_[index] : eq_constraints_[index];
146
147 if(grad_opt.size() > 0) {
148 val = constraint_func(x_real, grad_opt);
149 mapGradientToOpt(x_real, grad_opt);
150 } else {
151 Eigen::VectorXd dummy_grad(x_opt.size());
152 val = constraint_func(x_real, dummy_grad);
153 }
154 return val;
155 }
