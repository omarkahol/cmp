83 std::pair<Eigen::VectorXs, Eigen::VectorXs> operator*() const {
84 Eigen::Index fold_size = parent_.nObs_ / parent_.nSplits_;
85 Eigen::Index start = currentSplit_ * fold_size;
86 Eigen::Index end = (currentSplit_ == parent_.nSplits_ - 1)
87 ? parent_.nObs_ : start + fold_size;
88
89 // test set
90 Eigen::VectorXs test_indices = parent_.indices_.segment(start, end - start);
91
92 // train set
93 Eigen::VectorXs train_indices(parent_.nObs_ - (end - start));
94 Eigen::Index train_idx = 0;
95 for(Eigen::Index i = 0; i < parent_.nObs_; ++i) {
96 if(i < start || i >= end) {
97 train_indices(train_idx++) = parent_.indices_(i);
98 }
99 }
100
101 return {train_indices, test_indices};
102 }

156 }

165 Eigen::VectorXs operator()() {
166 Eigen::VectorXs sample_indices(nObsResample_);
167
168 if(withReplacement_) {
169 // Sample with replacement
170 for(size_t i = 0; i < nObsResample_; ++i) {
171 sample_indices(i) = dist_(rng_);
172 }
173 } else {
174 // Sample without replacement
175 // Shuffle all indices and take the first nObsResample_
176 Eigen::VectorXs shuffled_indices = indices_; // make a copy
177 std::shuffle(shuffled_indices.data(), shuffled_indices.data() + shuffled_indices.size(), rng_);
178 sample_indices = shuffled_indices.head(nObsResample_).eval();
179 }
180
181 return sample_indices;
182 }

268inline std::vector<Eigen::MatrixXd> selfCrossCorrelationFFT(
269 const Eigen::Ref<const Eigen::MatrixXd>& data) {
270 size_t nPoints = data.rows();
271 size_t nFeatures = data.cols();
272 size_t nFFT = 1;
273 while(nFFT < 2 * nPoints) nFFT <<= 1; // zero-pad to next power of 2
274
275 Eigen::FFT<double> fft;
276
277 // Precompute FFTs
278 std::vector<std::vector<std::complex<double>>> X(nFeatures, std::vector<std::complex<double>>(nFFT));
279 for(size_t i = 0; i < nFeatures; ++i) {
280 Eigen::VectorXd x = data.col(i);
281 x.array() -= x.mean();
282
283 std::vector<double> x_pad(nFFT, 0.0);
284 for(size_t t = 0; t < nPoints; ++t) x_pad[t] = x[t];
285
286 fft.fwd(X[i], x_pad);
287 }
288
289 // Compute cross-correlations for all lags
290 std::vector<Eigen::MatrixXd> corr(nPoints, Eigen::MatrixXd::Zero(nFeatures, nFeatures));
291
292 for(size_t i = 0; i < nFeatures; ++i) {
293 double var_i = data.col(i).array().square().sum() / nPoints;
294 for(size_t j = 0; j < nFeatures; ++j) {
295 double var_j = data.col(j).array().square().sum() / nPoints;
296
297 // Multiply FFTs
298 std::vector<std::complex<double>> R_fft(nFFT);
299 for(size_t k = 0; k < nFFT; ++k)
300 R_fft[k] = X[i][k] * std::conj(X[j][k]);
301
302 // Inverse FFT
303 std::vector<double> r_pad(nFFT, 0.0);
304 fft.inv(r_pad, R_fft);
305
306 // Normalize and store first nPoints lags
307 for(size_t lag = 0; lag < nPoints; ++lag)
308 corr[lag](i, j) = r_pad[lag] / (nPoints * std::sqrt(var_i * var_j));
309 }
310 }
311
312 return corr; // vector of size nPoints, each element = nFeatures x nFeatures

317class PairwiseDistanceStats {
318 private:
319 std::vector<double> distances;
320 double mean_val;
321 bool is_computed;
322
323 public:
324 PairwiseDistanceStats() : mean_val(0.0), is_computed(false) {}
325
330 void compute(const Eigen::MatrixXd& data) {
331 size_t n = data.rows();
332 if(n < 2) throw std::invalid_argument("Need at least 2 points to compute pairwise distances.");
333
334 // Number of strictly upper triangular elements: N * (N - 1) / 2
335 size_t num_pairs = n * (n - 1) / 2;
336 distances.resize(num_pairs);
337 mean_val = 0.0;
338
339 // OpenMP parallelization over the outer loop
340 #pragma omp parallel
341 {
342 // Thread-local sum prevents atomic locking overhead during the loop
343 double local_sum = 0.0;
344
345 // dynamic scheduling balances the triangular workload distribution
346 #pragma omp for schedule(dynamic)
347 for(size_t i = 0; i < n - 1; ++i) {
348 // Calculate the 1D flat array offset for row i
349 size_t offset = i * n - i * (i + 1) / 2 - i - 1;
350
351 for(size_t j = i + 1; j < n; ++j) {
352 // Eigen's .row() subtraction and .norm() naturally utilize SIMD
353 double dist = (data.row(i) - data.row(j)).norm();
354 distances[offset + j] = dist;
355 local_sum += dist;
356 }
357 }
358
359 // Safely accumulate the thread-local sums into the global mean
360 #pragma omp atomic
361 mean_val += local_sum;
362 }
363
364 mean_val /= static_cast<double>(num_pairs);
365 is_computed = true;
366 }
367
368 double mean() const {
369 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
370 return mean_val;
371 }
372
377 double quantile(double q) {
378 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
379 if(q < 0.0 || q > 1.0) throw std::invalid_argument("Quantile must be in [0, 1].");
380
381 size_t idx = static_cast<size_t>(q * (distances.size() - 1));
382
383 // std::nth_element provides an O(M) partial sort, which is significantly
384 // faster than an O(M log M) full std::sort.
385 std::nth_element(distances.begin(), distances.begin() + idx, distances.end());
386
387 return distances[idx];
388 }

395 std::vector<double> correlation_integral(const std::vector<double>& r_thresholds) const {
396 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
397 if(r_thresholds.empty()) return {};
398
399 size_t k = r_thresholds.size();
400 size_t m = distances.size();
401
402 // 1. Create a sorted copy of the thresholds for binary search tracking
403 std::vector<std::pair<double, size_t>> sorted_r(k);
404 for(size_t i = 0; i < k; ++i) {
405 sorted_r[i] = {r_thresholds[i], i};
406 }
407 std::sort(sorted_r.begin(), sorted_r.end(),
408 [](const auto & a, const auto & b) {
409 return a.first < b.first;
410 });
411
412 // Extract just the sorted values for std::upper_bound
413 std::vector<double> r_vals(k);
414 for(size_t i = 0; i < k; ++i) r_vals[i] = sorted_r[i].first;
415
416 // Global histogram bins
417 std::vector<size_t> global_counts(k + 1, 0);
418
419 // 2. OpenMP parallel histogram construction
420 #pragma omp parallel
421 {
422 // Thread-local histogram prevents atomic contention
423 std::vector<size_t> local_counts(k + 1, 0);
424
425 #pragma omp for schedule(static)
426 for(size_t i = 0; i < m; ++i) {
427 double d = distances[i];
428
429 // Binary search: O(log K) instead of O(K)
430 auto it = std::upper_bound(r_vals.begin(), r_vals.end(), d);
431 size_t idx = std::distance(r_vals.begin(), it);
432
433 // idx represents the first threshold strictly greater than d
434 local_counts[idx]++;
435 }
436
437 // Merge local histograms safely
438 #pragma omp critical
439 {
440 for(size_t j = 0; j <= k; ++j) {
441 global_counts[j] += local_counts[j];
442 }
443 }
444 }
445
446 // 3. Prefix Sum (Cumulative integration)
447 // If a distance falls in bin j, it is smaller than all thresholds >= j.
448 std::vector<double> C_r(k, 0.0);
449 size_t cumulative_sum = 0;
450
451 for(size_t j = 0; j < k; ++j) {
452 cumulative_sum += global_counts[j];
453
454 // Map the sorted result back to the original input order
455 size_t original_idx = sorted_r[j].second;
456 C_r[original_idx] = static_cast<double>(cumulative_sum) / static_cast<double>(m);
457 }
458
459 return C_r;
460 }
