132 std::pair<Eigen::VectorXs, Eigen::VectorXs> operator*() const {
133 Eigen::Index fold_size = parent_.nObs_ / parent_.nSplits_;
134 Eigen::Index start = currentSplit_ * fold_size;
135 Eigen::Index end = (currentSplit_ == parent_.nSplits_ - 1)
136 ? parent_.nObs_ : start + fold_size;
137
138 // test set
139 Eigen::VectorXs test_indices = parent_.indices_.segment(start, end - start);
140
141 // train set
142 Eigen::VectorXs train_indices(parent_.nObs_ - (end - start));
143 Eigen::Index train_idx = 0;
144 for(Eigen::Index i = 0; i < parent_.nObs_; ++i) {
145 if(i < start || i >= end) {
146 train_indices(train_idx++) = parent_.indices_(i);
147 }
148 }
149
150 return {train_indices, test_indices};
151 }

160 }

281 Eigen::VectorXs operator()() {
282 Eigen::VectorXs sample_indices(nObsResample_);
283
284 if(withReplacement_) {
285 // Sample with replacement
286 for(size_t i = 0; i < nObsResample_; ++i) {
287 sample_indices(i) = dist_(rng_);

291 // Shuffle all indices and take the first nObsResample_
292 Eigen::VectorXs shuffled_indices = indices_; // make a copy
293 std::shuffle(shuffled_indices.data(), shuffled_indices.data() + shuffled_indices.size(), rng_);
294 sample_indices = shuffled_indices.head(nObsResample_).eval();

417inline std::vector<Eigen::MatrixXd> selfCrossCorrelationFFT(
418 const Eigen::Ref<const Eigen::MatrixXd>& data) {
419 size_t nPoints = data.rows();
420 size_t nFeatures = data.cols();
421 size_t nFFT = 1;
422 while(nFFT < 2 * nPoints) nFFT <<= 1; // zero-pad to next power of 2
423
424 Eigen::FFT<double> fft;
425
426 // Precompute FFTs
427 std::vector<std::vector<std::complex<double>>> X(nFeatures, std::vector<std::complex<double>>(nFFT));
428 for(size_t i = 0; i < nFeatures; ++i) {
429 Eigen::VectorXd x = data.col(i);
430 x.array() -= x.mean();
431
432 std::vector<double> x_pad(nFFT, 0.0);
433 for(size_t t = 0; t < nPoints; ++t) x_pad[t] = x[t];
434
435 fft.fwd(X[i], x_pad);
436 }
437
438 // Compute cross-correlations for all lags
439 std::vector<Eigen::MatrixXd> corr(nPoints, Eigen::MatrixXd::Zero(nFeatures, nFeatures));
440
441 for(size_t i = 0; i < nFeatures; ++i) {
442 double var_i = data.col(i).array().square().sum() / nPoints;
443 for(size_t j = 0; j < nFeatures; ++j) {
444 double var_j = data.col(j).array().square().sum() / nPoints;
445
446 // Multiply FFTs
447 std::vector<std::complex<double>> R_fft(nFFT);
448 for(size_t k = 0; k < nFFT; ++k)
449 R_fft[k] = X[i][k] * std::conj(X[j][k]);
450
451 // Inverse FFT
452 std::vector<double> r_pad(nFFT, 0.0);
453 fft.inv(r_pad, R_fft);
454
455 // Normalize and store first nPoints lags
456 for(size_t lag = 0; lag < nPoints; ++lag)
457 corr[lag](i, j) = r_pad[lag] / (nPoints * std::sqrt(var_i * var_j));
458 }
459 }
460
461 return corr; // vector of size nPoints, each element = nFeatures x nFeatures
462};

486 PairwiseDistanceStats() : mean_val(0.0), is_computed(false) {}

500 void compute(const Eigen::MatrixXd& data) {
501 size_t n = data.rows();
502 if(n < 2) throw std::invalid_argument("Need at least 2 points to compute pairwise distances.");
503
504 // Number of strictly upper triangular elements: N * (N - 1) / 2
505 size_t num_pairs = n * (n - 1) / 2;
506 distances.resize(num_pairs);
507 mean_val = 0.0;
508
509 // OpenMP parallelization over the outer loop
510 #pragma omp parallel
511 {
512 // Thread-local sum prevents atomic locking overhead during the loop
513 double local_sum = 0.0;
514
515 // dynamic scheduling balances the triangular workload distribution
516 #pragma omp for schedule(dynamic)
517 for(size_t i = 0; i < n - 1; ++i) {
518 // Calculate the 1D flat array offset for row i
519 size_t offset = i * n - i * (i + 1) / 2 - i - 1;
520
521 for(size_t j = i + 1; j < n; ++j) {
522 // Eigen's .row() subtraction and .norm() naturally utilize SIMD
523 double dist = (data.row(i) - data.row(j)).norm();
524 distances[offset + j] = dist;
525 local_sum += dist;
526 }
527 }
528
529 // Safely accumulate the thread-local sums into the global mean
530 #pragma omp atomic
531 mean_val += local_sum;
532 }
533
534 mean_val /= static_cast<double>(num_pairs);
535 is_computed = true;
536 }
537
543 double mean() const {
544 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
545 return mean_val;
546 }
547
558 double quantile(double q) {
559 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
560 if(q < 0.0 || q > 1.0) throw std::invalid_argument("Quantile must be in [0, 1].");
561
562 size_t idx = static_cast<size_t>(q * (distances.size() - 1));
563
564 // std::nth_element provides an O(M) partial sort, which is significantly
565 // faster than an O(M log M) full std::sort.
566 std::nth_element(distances.begin(), distances.begin() + idx, distances.end());
567
568 return distances[idx];
569 }

588 std::vector<double> correlation_integral(const std::vector<double>& r_thresholds) const {
589 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
590 if(r_thresholds.empty()) return {};
591
592 size_t k = r_thresholds.size();
593 size_t m = distances.size();
594
595 // 1. Create a sorted copy of the thresholds for binary search tracking
596 std::vector<std::pair<double, size_t>> sorted_r(k);
597 for(size_t i = 0; i < k; ++i) {
598 sorted_r[i] = {r_thresholds[i], i};
599 }
600 std::sort(sorted_r.begin(), sorted_r.end(),
601 [](const auto & a, const auto & b) {
602 return a.first < b.first;
603 });
604
605 // Extract just the sorted values for std::upper_bound
606 std::vector<double> r_vals(k);
607 for(size_t i = 0; i < k; ++i) r_vals[i] = sorted_r[i].first;
608
609 // Global histogram bins
610 std::vector<size_t> global_counts(k + 1, 0);
611
612 // 2. OpenMP parallel histogram construction
613 #pragma omp parallel
614 {
615 // Thread-local histogram prevents atomic contention
616 std::vector<size_t> local_counts(k + 1, 0);
617
618 #pragma omp for schedule(static)
619 for(size_t i = 0; i < m; ++i) {
620 double d = distances[i];
621
622 // Binary search: O(log K) instead of O(K)
623 auto it = std::upper_bound(r_vals.begin(), r_vals.end(), d);
624 size_t idx = std::distance(r_vals.begin(), it);
625
626 // idx represents the first threshold strictly greater than d
627 local_counts[idx]++;
628 }
629
630 // Merge local histograms safely
631 #pragma omp critical
632 {
633 for(size_t j = 0; j <= k; ++j) {
634 global_counts[j] += local_counts[j];
635 }
636 }
637 }
638
639 // 3. Prefix Sum (Cumulative integration)
640 // If a distance falls in bin j, it is smaller than all thresholds >= j.
641 std::vector<double> C_r(k, 0.0);
642 size_t cumulative_sum = 0;
643
644 for(size_t j = 0; j < k; ++j) {
645 cumulative_sum += global_counts[j];
646
647 // Map the sorted result back to the original input order
648 size_t original_idx = sorted_r[j].second;
649 C_r[original_idx] = static_cast<double>(cumulative_sum) / static_cast<double>(m);
650 }
651
652 return C_r;
653 }
