\doxysubsection{Detailed Description}
Performs Global Sensitivity Analysis (GSA) using the Saltelli formulation of Sobol indices. 

\hypertarget{classcmp_1_1sobol_1_1_sobol_saltelli_autotoc_md15}{}\doxysubsubsubsection{\texorpdfstring{Mathematical Foundations}{Mathematical Foundations}}\label{classcmp_1_1sobol_1_1_sobol_saltelli_autotoc_md15}
Sobol sensitivity analysis decomposes the total variance $V(Y)$ of a model $Y = f(X_1, \dots, X_d)$ with independent inputs into contributions from individual inputs and their interactions\+: \[ V(Y) = \sum_{i=1}^d V_i + \sum_{1 \le i < j \le d} V_{ij} + \dots + V_{1\dots d} \]

Where\+:
\begin{DoxyItemize}
\item {\bfseries{First-\/\+Order Index (Main Effect)}}\+: Measures the fractional variance contribution of input $X_i$ alone\+: \[ S_i = \frac{V_i}{V(Y)} = \frac{\mathrm{Var}_{X_i}\left(\mathbb{E}_{\mathbf{X}_{\sim i}}[Y | X_i]\right)}{\mathrm{Var}(Y)} \]
\item {\bfseries{Second-\/\+Order Index (Interaction Effect)}}\+: Measures the interaction effect of inputs $X_i$ and $X_j$\+: \[ S_{ij} = \frac{V_{ij}}{V(Y)} = \frac{\mathrm{Var}_{X_i, X_j}\left(\mathbb{E}_{\mathbf{X}_{\sim i,j}}[Y | X_i, X_j]\right) - V_i - V_j}{\mathrm{Var}(Y)} \]
\item {\bfseries{Total-\/\+Order Index}}\+: Measures the total effect of $X_i$ including all its interactions\+: \[ S_{Ti} = 1 - \frac{\mathrm{Var}_{\mathbf{X}_{\sim i}}\left(\mathbb{E}_{X_i}[Y | \mathbf{X}_{\sim i}]\right)}{\mathrm{Var}(Y)} \]
\end{DoxyItemize}\hypertarget{classcmp_1_1sobol_1_1_sobol_saltelli_autotoc_md16}{}\doxysubsubsubsection{\texorpdfstring{Implementation Algorithms (Saltelli 2010 Scheme)}{Implementation Algorithms (Saltelli 2010 Scheme)}}\label{classcmp_1_1sobol_1_1_sobol_saltelli_autotoc_md16}

\begin{DoxyEnumerate}
\item {\bfseries{Grid Generation}}\+: We draw two independent sample matrices $\mathbf{A}$ and $\mathbf{B}$ of size $N \times d$. We construct $d$ hybrid matrices $\mathbf{A}_B^i$ (where column $i$ is from $\mathbf{B}$, rest from $\mathbf{A}$) and $d(d-1)/2$ hybrid matrices $\mathbf{A}_B^{ij}$ (where columns $i$ and $j$ are from $\mathbf{B}$, rest from $\mathbf{A}$).
\item {\bfseries{Evaluation}}\+: Evaluate the model outputs\+: $\mathbf{y}_A$, $\mathbf{y}_B$, $\mathbf{y}_{AB}^i$, and $\mathbf{y}_{AB}^{ij}$.
\item {\bfseries{Estimator Formulations}}\+:
\begin{DoxyItemize}
\item First-\/\+Order\+: \[ S_i = \frac{\frac{1}{N} \sum_{k=1}^N y_B^{(k)} (y_{AB}^{(i)(k)} - y_A^{(k)})}{\mathrm{Var}(Y)} \]
\item Total-\/\+Order\+: \[ S_{Ti} = \frac{\frac{1}{2N} \sum_{k=1}^N (y_A^{(k)} - y_{AB}^{(i)(k)})^2}{\mathrm{Var}(Y)} \]
\end{DoxyItemize}
\end{DoxyEnumerate}\hypertarget{classcmp_1_1sobol_1_1_sobol_saltelli_autotoc_md17}{}\doxysubsubsubsection{\texorpdfstring{Constraints \& Invariants}{Constraints \& Invariants}}\label{classcmp_1_1sobol_1_1_sobol_saltelli_autotoc_md17}

\begin{DoxyItemize}
\item {\bfseries{Sample Independence}}\+: The input matrices $\mathbf{A}$ and $\mathbf{B}$ must be statistically independent.
\item {\bfseries{Computational Complexity}}\+: The number of model evaluations required is $N(d + 2)$ for first and total order, and $N(2d + 2 + d(d-1)/2)$ if second-\/order interaction indices are requested. 
\end{DoxyItemize}
