20inline double erfinv(float x) {
21 double tt1, tt2, lnx, sgn;
22 sgn = (x < 0) ? -1.0f : 1.0f;
23 x = 1 - x * x;
24 lnx = std::log(x);
25 tt1 = 2 / (M_PI * 0.147) + 0.5f * lnx;
26 tt2 = 1 / (0.147) * lnx;
27 return (sgn * std::sqrt(-tt1 + std::sqrt(tt1 * tt1 - tt2)));
28}

310 public:
311 BetaDistribution(double alpha, double beta)
312 : alpha_(alpha), beta_(beta),
313 distGammaAlpha_(alpha, 1.0),

322 double logPDF(const double &x) const {
323 // Beta distribution is strictly defined on the interval (0, 1)
324 if(x <= 0.0 || x >= 1.0) return -std::numeric_limits<double>::infinity();
325
326 // Calculate the log Beta function: ln(B(a, b)) = ln(Gamma(a)) + ln(Gamma(b)) - ln(Gamma(a+b))
327 double logBetaFunc = std::lgamma(alpha_) + std::lgamma(beta_) - std::lgamma(alpha_ + beta_);

329 return (alpha_ - 1.0) * std::log(x) + (beta_ - 1.0) * std::log(1.0 - x) - logBetaFunc;
330 }

333 if(x <= 0.0 || x >= 1.0) return 0.0;

335 }

337 double ddLogPDF(const double &x) const {
338 if(x <= 0.0 || x >= 1.0) return 0.0;
339 return -(alpha_ - 1.0) / std::pow(x, 2) - (beta_ - 1.0) / std::pow(1.0 - x, 2);
340 }

343 return 0.0; // To-do: Requires regularized incomplete beta function

352 double y = distGammaBeta_(rng);
353
354 // Prevent division by zero in the extremely rare case both evaluate to 0.0

571 static double logPDF(const Eigen::Ref<const Eigen::VectorXd> &res, const Eigen::LDLT<Eigen::MatrixXd> &ldltDecomposition) {
572 Eigen::VectorXd alpha = ldltDecomposition.solve(res);
573 return -0.5 * res.dot(alpha) - 0.5 * (ldltDecomposition.vectorD().array().abs().log()).sum() - 0.5 * res.size() * std::log(2 * M_PI);
574 }
575 static double dLogPDF(const Eigen::Ref<const Eigen::VectorXd> &res, const Eigen::LDLT<Eigen::MatrixXd> &ldltDecomposition, const Eigen::Ref<const Eigen::MatrixXd> &cov_gradient, const Eigen::Ref<const Eigen::VectorXd> &mean_gradient) {
576 Eigen::VectorXd alpha = ldltDecomposition.solve(res);
577 Eigen::MatrixXd alpha_alpha_t = alpha * alpha.transpose();
578 return 0.5 * (alpha_alpha_t * cov_gradient - ldltDecomposition.solve(cov_gradient)).trace() + res.dot(ldltDecomposition.solve(mean_gradient));
579 }
580 static double ddLogPDF(const Eigen::Ref<const Eigen::VectorXd> &res, const Eigen::LDLT<Eigen::MatrixXd> &ldltDecomposition, const Eigen::Ref<const Eigen::MatrixXd> &cov_gradient_l, const Eigen::Ref<const Eigen::MatrixXd> &cov_gradient_k, const Eigen::Ref<const Eigen::MatrixXd> &cov_hessian) {
581 Eigen::VectorXd alpha = ldltDecomposition.solve(res);
582 Eigen::MatrixXd alpha_alpha_t = alpha * alpha.transpose();
583 auto a_l = ldltDecomposition.solve(cov_gradient_l);
584 auto a_k = ldltDecomposition.solve(cov_gradient_k);
585 auto sym_tens = 0.5 * ((a_l * alpha_alpha_t) + (a_l * alpha_alpha_t).transpose());
586 double H1 = (alpha_alpha_t*cov_hessian).trace();
587 double H2 = (ldltDecomposition.solve(cov_hessian)).trace();
588 double H3 = (sym_tens * cov_gradient_k).trace();
589 double H4 = (a_l * a_k).trace();
590 return 0.5 * H1 - 0.5 * H2 - H3 + 0.5 * H4;
591 }

691 Eigen::VectorXd d = ldltDecomposition.vectorD();
692 Eigen::MatrixXd D = d.asDiagonal();
693 Eigen::MatrixXd L = ldltDecomposition.matrixL().toDenseMatrix();
694 Eigen::MatrixXd cov = L * D * L.transpose();
695 ldltDecomposition_.compute(cov);
696 }
697 MultivariateStudentDistribution(const Eigen::Ref<const Eigen::VectorXd> &mean, const Eigen::Ref<const Eigen::MatrixXd> &cov, double nu)
698 : mean_(mean), dofs_(nu), distN_(0., 1.) {
699 ldltDecomposition_.compute(cov);

702
703 static double logPDF(const Eigen::Ref<const Eigen::VectorXd> &res, const Eigen::LDLT<Eigen::MatrixXd> &ldltDecomposition, const double &nu) {
704 size_t dim = res.size();
705 Eigen::VectorXd alpha = ldltDecomposition.solve(res);
706 double quad = res.dot(alpha);
707 double logDet = ldltDecomposition.vectorD().array().log().sum();
708 double logGammaTerm = std::lgamma(0.5 * (nu + dim)) - std::lgamma(0.5 * nu);
709 double logNormTerm = -0.5 * dim * std::log(nu * M_PI);
710 double logDetTerm = -0.5 * logDet;
711 double logQuadTerm = -0.5 * (nu + dim) * std::log(1 + quad / nu);
712 return logGammaTerm + logNormTerm + logDetTerm + logQuadTerm;
713 }
714
715 double logPDF(const Eigen::Ref<const Eigen::VectorXd> &x) const {
716 return logPDF(mean_ - x, ldltDecomposition_, dofs_);
717 }
718 double logJumpPDF(const Eigen::Ref<const Eigen::VectorXd> &jump) {
719 return logPDF(jump, ldltDecomposition_, dofs_);
720 }
721 double squaredMahalanobis(const Eigen::Ref<const Eigen::VectorXd> &jump) const {
722 return jump.dot(ldltDecomposition_.solve(jump));
723 }
724
725 Eigen::VectorXd sample(std::default_random_engine &rng, const double &gamma = 1.0) {
726 Eigen::VectorXd z = Eigen::VectorXd::Zero(mean_.size());
727 for(size_t i = 0; i < mean_.size(); i++) {
728 z(i) = distN_(rng);
729 double chi = 0.0;
730 for(size_t j = 0; j < dofs_; j++) {
731 chi += std::pow(distN_(rng), 2);
732 }
733 z(i) = std::sqrt(std::abs(ldltDecomposition_.vectorD()(i))) * z(i) / std::sqrt(chi / dofs_);
734 }
735 return mean_ + (ldltDecomposition_.transpositionsP().transpose() * (ldltDecomposition_.matrixL() * z)) * gamma;
736 }

739 return mean_;

877 private:
878 Eigen::VectorXd mean_;
879 double kappa_;
880 double nu_;
881 size_t dim_;

884 std::normal_distribution<double> distN_{0., 1.};
885
886 public:
887
888 NormalInverseWishartDistribution(
889 const Eigen::Ref<const Eigen::VectorXd> &mean,
890 double kappa, double nu,
891 const Eigen::Ref<const Eigen::MatrixXd> &psi
892 ) : mean_(mean), kappa_(kappa), nu_(nu - mean.size() + 1), dim_(mean.size()) {
893 if(psi.rows() != psi.cols() || psi.rows() != dim_) throw std::invalid_argument("Scale matrix Ψ must be square and match the dimensionality of the mean");
894 if(kappa_ <= 0) throw std::invalid_argument("κ must be positive");
895 if(nu <= dim_ - 1) throw std::invalid_argument("ν must be greater than dimension - 1");
896
897 double scaling = (kappa_ + 1.0) / (kappa_ * nu_);
898 Eigen::MatrixXd scaledPsi = scaling * psi;
899 covLDLT_.compute(scaledPsi);
900 if(covLDLT_.info() != Eigen::Success) throw std::invalid_argument("Scaled covariance matrix must be positive definite");
901 logDeterminant_ = covLDLT_.vectorD().array().log().sum();

905 double logPDF(const Eigen::Ref<const Eigen::VectorXd> &x) const {
906 Eigen::VectorXd res = x - mean_;

908 }
909
910 Eigen::VectorXd sample(std::default_random_engine &rng) {
911 const auto &L = covLDLT_.matrixL();
912 const auto &P = covLDLT_.transpositionsP();
913 const Eigen::VectorXd D = covLDLT_.vectorD();
914
915 Eigen::VectorXd z = Eigen::VectorXd::Zero(dim_);
916 for(size_t i = 0; i < dim_; ++i) {
917 z(i) = distN_(rng) * std::sqrt(std::abs(D(i)));
918 }
919 Eigen::VectorXd y = P.transpose() * (L * z);
920
921 std::chi_squared_distribution<double> distChi(nu_);
922 double s = distChi(rng);
923 return mean_ + y / std::sqrt(s / nu_);
924 }

928 }

930 return kappa_;
931 }

969 static NormalInverseWishartDistribution empiricalPrior(const Eigen::MatrixXd &data, double expected_clusters, double kappa0) {
970
971 int D = static_cast<int>(data.cols());
972 double N = static_cast<double>(data.rows());
973
974 // 1. Compute empirical global mean (1 x D) -> Convert to Column Vector (D x 1)
975 Eigen::VectorXd mu0 = data.colwise().mean().transpose();
976
977 // 2. Set degrees of freedom to the weakest valid setting
978 double nu0 = D + 2.0;
979
980 // 3. Compute empirical covariance matrix (D x D)
981 // Centering the data: subtract the mean row from every row
982 Eigen::MatrixXd centered = data.rowwise() - data.colwise().mean();
983
984 // Covariance = (Centered^T * Centered) / (N - 1)
985 Eigen::MatrixXd global_cov = (centered.transpose() * centered) / (N - 1.0);
986
987 // 4. Scale Psi0 (the scale matrix) based on expected cluster density
988 // We multiply by (nu0 - D - 1) to ensure the expected value of the prior covariance
989 // matches the fraction of the global variance exactly.
990 double scale_factor = (nu0 - static_cast<double>(D) - 1.0) / expected_clusters;
991 Eigen::MatrixXd Psi0 = global_cov * scale_factor;
992
993 // Fix potential numerical edge-cases where columns have zero variance
994 Psi0.diagonal().array() += 1e-6;
995
996 // 5. Return the constructed instance
997 // (Assuming your constructor takes mu0, kappa0, nu0, and Psi0)
998 return NormalInverseWishartDistribution(mu0, kappa0, nu0, Psi0);
999 }
