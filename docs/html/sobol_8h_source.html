55 double getSecondOrder(size_t i, size_t j) const {
56 if(secondOrder.size() == 0) throw std::logic_error("Second order Sobol indices were not computed.");
57 if(i >= firstOrder.size() || j >= firstOrder.size()) throw std::out_of_range("Index out of range");
58 if(i == j) throw std::invalid_argument("Indices must be different for second order Sobol index.");
