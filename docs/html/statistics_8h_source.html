114 std::pair<Eigen::VectorXs, Eigen::VectorXs> operator*() const {
115 Eigen::Index fold_size = parent_.nObs_ / parent_.nSplits_;
116 Eigen::Index start = currentSplit_ * fold_size;
117 Eigen::Index end = (currentSplit_ == parent_.nSplits_ - 1)
118 ? parent_.nObs_ : start + fold_size;
119
120 // test set
121 Eigen::VectorXs test_indices = parent_.indices_.segment(start, end - start);
122
123 // train set
124 Eigen::VectorXs train_indices(parent_.nObs_ - (end - start));
125 Eigen::Index train_idx = 0;
126 for(Eigen::Index i = 0; i < parent_.nObs_; ++i) {
127 if(i < start || i >= end) {
128 train_indices(train_idx++) = parent_.indices_(i);
129 }
130 }
131
132 return {train_indices, test_indices};
133 }

216 if(n_obs < 1) {
217 throw std::invalid_argument("Number of observations must be at least 1.");
218 }
219 if(n_obs_resample < 1) {
220 throw std::invalid_argument("Number of resampled observations must be at least 1.");
221 }
222 if(!withReplacement_ && n_obs_resample > n_obs) {
223 throw std::invalid_argument("Number of resampled observations cannot be greater than number of observations when sampling without replacement.");
224 }
225
226 // Generate indices 0..nObs-1
227 indices_.resize(nObs_);
228 for(size_t i = 0; i < nObs_; ++i) {
229 indices_(i) = i;
230 }
231 }
232
233 Eigen::VectorXs operator()() {
234 Eigen::VectorXs sample_indices(nObsResample_);
235
236 if(withReplacement_) {
237 // Sample with replacement
238 for(size_t i = 0; i < nObsResample_; ++i) {
239 sample_indices(i) = dist_(rng_);
240 }
241 } else {
242 // Sample without replacement
243 // Shuffle all indices and take the first nObsResample_
244 Eigen::VectorXs shuffled_indices = indices_; // make a copy
245 std::shuffle(shuffled_indices.data(), shuffled_indices.data() + shuffled_indices.size(), rng_);
246 sample_indices = shuffled_indices.head(nObsResample_).eval();
247 }
248
249 return sample_indices;
250 }

336inline std::vector<Eigen::MatrixXd> selfCrossCorrelationFFT(
337 const Eigen::Ref<const Eigen::MatrixXd>& data) {
338 size_t nPoints = data.rows();
339 size_t nFeatures = data.cols();
340 size_t nFFT = 1;
341 while(nFFT < 2 * nPoints) nFFT <<= 1; // zero-pad to next power of 2
342
343 Eigen::FFT<double> fft;
344
345 // Precompute FFTs
346 std::vector<std::vector<std::complex<double>>> X(nFeatures, std::vector<std::complex<double>>(nFFT));
347 for(size_t i = 0; i < nFeatures; ++i) {
348 Eigen::VectorXd x = data.col(i);
349 x.array() -= x.mean();
350
351 std::vector<double> x_pad(nFFT, 0.0);
352 for(size_t t = 0; t < nPoints; ++t) x_pad[t] = x[t];
353
354 fft.fwd(X[i], x_pad);
355 }
356
357 // Compute cross-correlations for all lags
358 std::vector<Eigen::MatrixXd> corr(nPoints, Eigen::MatrixXd::Zero(nFeatures, nFeatures));
359
360 for(size_t i = 0; i < nFeatures; ++i) {
361 double var_i = data.col(i).array().square().sum() / nPoints;
362 for(size_t j = 0; j < nFeatures; ++j) {
363 double var_j = data.col(j).array().square().sum() / nPoints;
364
365 // Multiply FFTs
366 std::vector<std::complex<double>> R_fft(nFFT);
367 for(size_t k = 0; k < nFFT; ++k)
368 R_fft[k] = X[i][k] * std::conj(X[j][k]);
369
370 // Inverse FFT
371 std::vector<double> r_pad(nFFT, 0.0);
372 fft.inv(r_pad, R_fft);
373
374 // Normalize and store first nPoints lags
375 for(size_t lag = 0; lag < nPoints; ++lag)
376 corr[lag](i, j) = r_pad[lag] / (nPoints * std::sqrt(var_i * var_j));
377 }
378 }
379
380 return corr; // vector of size nPoints, each element = nFeatures x nFeatures
381};

385class PairwiseDistanceStats {
386 private:
387 std::vector<double> distances;
388 double mean_val;
389 bool is_computed;
390
391 public:
392 PairwiseDistanceStats() : mean_val(0.0), is_computed(false) {}
393
398 void compute(const Eigen::MatrixXd& data) {
399 size_t n = data.rows();
400 if(n < 2) throw std::invalid_argument("Need at least 2 points to compute pairwise distances.");
401
402 // Number of strictly upper triangular elements: N * (N - 1) / 2
403 size_t num_pairs = n * (n - 1) / 2;
404 distances.resize(num_pairs);
405 mean_val = 0.0;
406
407 // OpenMP parallelization over the outer loop
408 #pragma omp parallel
409 {
410 // Thread-local sum prevents atomic locking overhead during the loop
411 double local_sum = 0.0;
412
413 // dynamic scheduling balances the triangular workload distribution
414 #pragma omp for schedule(dynamic)
415 for(size_t i = 0; i < n - 1; ++i) {
416 // Calculate the 1D flat array offset for row i
417 size_t offset = i * n - i * (i + 1) / 2 - i - 1;
418
419 for(size_t j = i + 1; j < n; ++j) {
420 // Eigen's .row() subtraction and .norm() naturally utilize SIMD
421 double dist = (data.row(i) - data.row(j)).norm();
422 distances[offset + j] = dist;
423 local_sum += dist;
424 }
425 }
426
427 // Safely accumulate the thread-local sums into the global mean
428 #pragma omp atomic
429 mean_val += local_sum;
430 }
431
432 mean_val /= static_cast<double>(num_pairs);
433 is_computed = true;
434 }
435
436 double mean() const {
437 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
438 return mean_val;
439 }
440
445 double quantile(double q) {
446 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
447 if(q < 0.0 || q > 1.0) throw std::invalid_argument("Quantile must be in [0, 1].");
448
449 size_t idx = static_cast<size_t>(q * (distances.size() - 1));
450
451 // std::nth_element provides an O(M) partial sort, which is significantly
452 // faster than an O(M log M) full std::sort.
453 std::nth_element(distances.begin(), distances.begin() + idx, distances.end());
454
455 return distances[idx];
456 }
457
463 std::vector<double> correlation_integral(const std::vector<double>& r_thresholds) const {
464 if(!is_computed) throw std::runtime_error("Distances not yet computed.");
465 if(r_thresholds.empty()) return {};
466
467 size_t k = r_thresholds.size();
468 size_t m = distances.size();
469
470 // 1. Create a sorted copy of the thresholds for binary search tracking
471 std::vector<std::pair<double, size_t>> sorted_r(k);
472 for(size_t i = 0; i < k; ++i) {
473 sorted_r[i] = {r_thresholds[i], i};
474 }
475 std::sort(sorted_r.begin(), sorted_r.end(),
476 [](const auto & a, const auto & b) {
477 return a.first < b.first;
478 });
479
480 // Extract just the sorted values for std::upper_bound
481 std::vector<double> r_vals(k);
482 for(size_t i = 0; i < k; ++i) r_vals[i] = sorted_r[i].first;
483
484 // Global histogram bins
485 std::vector<size_t> global_counts(k + 1, 0);
486
487 // 2. OpenMP parallel histogram construction
488 #pragma omp parallel
489 {
490 // Thread-local histogram prevents atomic contention
491 std::vector<size_t> local_counts(k + 1, 0);
492
493 #pragma omp for schedule(static)
494 for(size_t i = 0; i < m; ++i) {
495 double d = distances[i];
496
497 // Binary search: O(log K) instead of O(K)
498 auto it = std::upper_bound(r_vals.begin(), r_vals.end(), d);
499 size_t idx = std::distance(r_vals.begin(), it);
500
501 // idx represents the first threshold strictly greater than d
502 local_counts[idx]++;
503 }
504
505 // Merge local histograms safely
506 #pragma omp critical
507 {
508 for(size_t j = 0; j <= k; ++j) {
509 global_counts[j] += local_counts[j];
510 }
511 }
512 }
513
514 // 3. Prefix Sum (Cumulative integration)
515 // If a distance falls in bin j, it is smaller than all thresholds >= j.
516 std::vector<double> C_r(k, 0.0);
517 size_t cumulative_sum = 0;
518
519 for(size_t j = 0; j < k; ++j) {
520 cumulative_sum += global_counts[j];
521
522 // Map the sorted result back to the original input order
523 size_t original_idx = sorted_r[j].second;
524 C_r[original_idx] = static_cast<double>(cumulative_sum) / static_cast<double>(m);
525 }
526
527 return C_r;
528 }
