277 void updateAlpha() {
278 double K = static_cast<double>(clusters_.size());
279 double N = static_cast<double>(nPoints_);
280
281 // Extract prior parameters from your distribution object
282 double a = alphaPrior_.getAlpha();
283 double b = alphaPrior_.getBeta();
284
285 // 1. Instantiate the Beta distribution and sample eta
286 cmp::distribution::BetaDistribution beta_dist(alpha_ + 1.0, N);
287 double eta = beta_dist.sample(rng_);
288
289 // 2. Calculate the weights for the Gamma mixture
290 double weight1 = (a + K - 1.0) / (N * (b - std::log(eta)));
291 double pi_eta = weight1 / (weight1 + 1.0);
292
293 // 3. Sample the new alpha using your custom cmp::distribution classes
294 double u = distU_(rng_);
295 double updated_beta = b - std::log(eta); // The rate parameter
296
297 if(u < pi_eta) {
298 // Gamma(a + K, b - ln(eta))
299 cmp::distribution::GammaDistribution new_gamma(a + K, updated_beta);
300 alpha_ = new_gamma.sample(rng_);
301 } else {
302 // Gamma(a + K - 1, b - ln(eta))
303 cmp::distribution::GammaDistribution new_gamma(a + K - 1.0, updated_beta);
304 alpha_ = new_gamma.sample(rng_);
305 }
306
307 // Safety bounds just in case of severe numerical underflow
308 if(alpha_ < 1e-5) alpha_ = 1e-5;
309 }
