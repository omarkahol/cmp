105// We implement a full bandwidth based on the Cholesky decomposition
106class FullBandwidth : public Bandwidth {
107
108 private:
109 Eigen::LLT<Eigen::MatrixXd> L_;
110 double det_;
111 size_t dim_;
112
113 public:
114 FullBandwidth() = delete;
115
116 FullBandwidth(const Eigen::MatrixXd &LLT) : dim_(LLT.rows()) {
117 if(LLT.rows() != LLT.cols()) {
118 throw std::invalid_argument("Full bandwidth expects a square matrix.");
119 }
120 L_ = Eigen::LLT<Eigen::MatrixXd>(LLT);
121 if(L_.info() != Eigen::Success) {
122 throw std::invalid_argument("The provided matrix is not positive definite.");
123 }
124
125 // Compute determinant as product of diagonal entries of the Cholesky factor
126 // directly to avoid hitting Eigen internals that trigger deprecated enum
127 // bitwise operations.
128 det_ = 1.0;
129 for(size_t i = 0; i < static_cast<size_t>(L_.matrixL().rows()); ++i) {
130 det_ *= L_.matrixL()(i, i);
131 }
132 };
133
134 double determinant() const override;
135
136 size_t size() const {
137 return dim_;
138 };
139
140 std::pair<size_t, size_t> indexToRowCol(const size_t &i) const;
