42template <typename BasisType>
43class Quadrature1D {
44 public:
45 Eigen::VectorXd nodes;
46 Eigen::VectorXd weights;
47
48 Quadrature1D(int numPoints) {
49 if(numPoints < 1) throw std::invalid_argument("Points must be >= 1");
50
51 nodes.resize(numPoints);
52 weights.resize(numPoints);
53
54 if(numPoints == 1) {
55 nodes(0) = 0.0;
56 weights(0) = 1.0;
57 return;
58 }
59
60 Eigen::MatrixXd J = BasisType::getJacobiMatrix(numPoints);
61 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> solver(J);
62 nodes = solver.eigenvalues();
63 Eigen::MatrixXd evecs = solver.eigenvectors();
64
65 for(int i = 0; i < numPoints; ++i) {
66 weights(i) = evecs(0, i) * evecs(0, i);
67 }
68 }
69
70 double integrate(std::function<double(double)> f) const {
71 double result = 0.0;
72 for(int i = 0; i < nodes.size(); ++i) {
73 result += weights(i) * f(nodes(i));
74 }
75 return result;
76 }
77};

103template <typename BasisType>
104class TensorIntegrator {
105 public:
106 Eigen::MatrixXd gridNodes;
107 Eigen::VectorXd gridWeights;
108
109 TensorIntegrator(int dim, int pointsPerDim) {
110 Quadrature1D<BasisType> gq1D(pointsPerDim);
111
112 int totalPoints = std::pow(pointsPerDim, dim);
113 gridNodes.resize(totalPoints, dim);
114 gridWeights.resize(totalPoints);
115
116 for(int i = 0; i < totalPoints; ++i) {
117 int temp = i;
118 double jointWeight = 1.0;
119
120 for(int d = 0; d < dim; ++d) {
121 int idx = temp % pointsPerDim;
122 gridNodes(i, d) = gq1D.nodes(idx);
123 jointWeight *= gq1D.weights(idx);
124 temp /= pointsPerDim;
125 }
126 gridWeights(i) = jointWeight;
127 }
128 }
129
135 double integrate(const std::function<double(const Eigen::VectorXd&)> &f) const {
136 double result = 0.0;
137 for(int i = 0; i < gridNodes.rows(); ++i) {
138 result += gridWeights(i) * f(gridNodes.row(i));
139 }
140 return result;
141 }
